-- ── Logic ────────────────────────────────────────────────────────────────

local enabled = true

Toggle.MouseButton1Click:Connect(function()
    enabled = not enabled

    if enabled then
        Toggle.Text = "ON"
        Toggle.BackgroundColor3 = Color3.fromRGB(22,22,24)
        ToggleStroke.Color = Color3.fromRGB(230,230,240)
    else
        Toggle.Text = "OFF"
        Toggle.BackgroundColor3 = Color3.fromRGB(10,10,12)
        ToggleStroke.Color = Color3.fromRGB(120,120,130)
    end
end)

-- Speed boost (client-side velocity override)
RunService.Heartbeat:Connect(function(deltaTime)
    if not enabled then return end
    if not Humanoid then return end
    if not HumanoidRootPart then return end

    local moveDir = Humanoid.MoveDirection

    if moveDir.Magnitude > 0 and Humanoid.WalkSpeed < 25 then
        -- Very aggressive ~28 studs/sec in move direction (ignores walkspeed cap)
        HumanoidRootPart.AssemblyLinearVelocity = Vector3.new(
            moveDir.X * 28,
            HumanoidRootPart.AssemblyLinearVelocity.Y,  -- preserve Y (jump/fall)
            moveDir.Z * 28
        )
    end
end)

-- Pretty rotating gradients
RunService.Heartbeat:Connect(function(deltaTime)
    Gradient.Rotation = (Gradient.Rotation + 90 * deltaTime) % 360
    StrokeGradient.Rotation = (StrokeGradient.Rotation + 140 * deltaTime) % 360
end)
